Aufgabe 1.6

SELECT auftrnr, sum(preis*anzahl) AS preis
FROM AUFTRAGSPOS

JOIN preisliste
ON auftragspos.teilenr=preisliste.teilenr AND auftragspos.farbe=preisliste.farbe

GROUP BY auftrnr

ORDER BY preis ASC
